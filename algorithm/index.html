                var allNum = [1,2,3,4,5,6,7];
                var sort = function(nums,one){
                    var low = 0, height = nums.length -1;
                    while(low <= height){
                        debugger;
                        var mid = Math.floor((low + height)/2);
                        if(one === nums[mid]){
                            console.info("第" + mid +"个数");
                            return nums[mid];
                        }
                        if(one > nums[mid]){
                            low = mid + 1;
                        }
                        if(one < nums[mid]){
                            height = mid-1;
                        }
                    }
                }
